The characteristic equation for a matrix $\vec{A}$ is 
\begin{align}
\label{eq:chareq}
	f\brak{\lambda} =\mydet{\vec{A}-\lambda \vec{I}}=0
\end{align}
\item Cayley-Hamilton theorem
\begin{align}
\label{eq:cayley}
	f\brak{\lambda}=f\brak{\vec{A}} = 0
\end{align}
